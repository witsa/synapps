import { assert } from '@ember/debug';
import { tracked } from '@glimmer/tracking';
import accessoryDefs from 'synapps-ui/accessories/defs';
import DataContextual from 'synapps-ui/behaviors/data-contextual';
import Destroyable from 'synapps-ui/behaviors/destroyable';
import SourceOfInternalBinding from 'synapps-ui/behaviors/source-of-internal-binding';
import WithProperties from 'synapps-ui/behaviors/with-properties';

@WithProperties
@DataContextual('actor')
@SourceOfInternalBinding
@Destroyable
class BaseAccessoryObject extends Object {}

/**
 * Accessoire
 * @namespace Accessory
 */

/**
 * Classe de base des accessoires.
 * Elle est accessible par n'importe quel acteur dans le champ [BaseAccessory]{@link Actor.BaseActor#BaseAccessory}.
 *
 * Un accessoire permet de définir des comportements à reproduire sur des acteurs en se greffant sur leurs évènements et leur cycle de vie.
 *
 * Un accessoire hérite du contexte de donnée de l'acteur auquel il appartient.
 * @class Accessory.BaseAccessory
 * @extends Behavior.WithProperties
 * @extends Behavior.DataContextual
 * @extends Behavior.SourceOfInternalBinding
 * @extends Behavior.Destroyable
 */
export default class BaseAccessory extends BaseAccessoryObject {

  static prototypeSetup;

  static accessoryDefs = accessoryDefs;

  /**
   * Enregistre la définition de l'accessoire dans le dictionnaire des accessoires pour le rendre accessible à la création par setup JSON.
   *
   * @static
   * @name Accessory.BaseAccessory#registerInAccessoryDefs
   * @param {Object} definition
   * @param {string} definition.type - La clé unique du type de l'accessoire (ex: "data-source/wos-relative-variable");
   * @param {string} definition.prototypeSetup - Object qui contient les définitions des propriétés de l'accessoire.
   * @param {string} [definition.platform=native] - Nom de la plateforme à laquelle l'accessoire appartient.
   */
  static registerInAccessoryDefs({ type, prototypeSetup, platform = 'native' }) {
    assert('[BaseAccessory][registerInAccessoryDefs] type is required.', type);
    assert('[BaseAccessory][registerInAccessoryDefs] prototypeSetup is required.', prototypeSetup);
    this.type = type;
    this.platform = platform;
    this.prototypeSetup = prototypeSetup;
    this.accessoryDefs.set(type, this);
  }

  _actor;
  _setup;

  @tracked
  _key;

  /**
   * Obtient le chemin unique vers l'accessoire.
   * @member Accessory.BaseAccessory#uniqPath
   * @readonly
   * @type {String}
   */
  get uniqPath() {
    return `${this.actor.uniqPath}.accessories.${this.key}`;
  }

  constructor({ key, actor, setup }) {
    super();
    assert('[BaseAccessory][constructor] key is required.', key);
    assert('[BaseAccessory][constructor] actor is required.', actor);
    assert('[BaseAccessory][constructor] setup is required.', setup);

    this._init({ key, actor, setup });
  }

  _init({ key, actor, setup }) {
    this._key = key;
    this._actor = actor;
    this._setup = setup;

    this._extractPropertiesFromSetup(setup, this.constructor.prototypeSetup);
    // logger.debug('[BaseAccessory][_init]', this.uniqPath);
  }

  _doAfterPropertyChanged({ propertyPath, newValue, name, ...args }) {
    const accessoryPropertyPath = `accessories.${this.key}.${propertyPath}`;
    this.actor.runEvent('onPropertyChanged', { propertyPath: accessoryPropertyPath, newValue, ...args });
    this.actor._runPropertyEvent(accessoryPropertyPath, 'onValueChanged', newValue, { newValue, propertyPath: accessoryPropertyPath, ...args });
    this._invalidateProperties(name);
  }

  /**
   * Obtient l'[acteur]{@link Actor.BaseActor} qu'il décore.
   * @member Accessory.BaseAccessory#actor
   * @type {Actor.BaseActor}
   * @readonly
   */
  get actor() {
    return this._actor;
  }

  /**
   * Obtient la clé unique de l'accessoire qui l'identifie parmi les autres [accessoires]{@link Actor.BaseActor#accessories} de l'[acteur]{@link Actor.BaseActor}.
   * @member Accessory.BaseAccessory#key
   * @readonly
   * @type {String}
   */
  get key() {
    return this._key;
  }

  get scene() {
    return this.actor.scene;
  }

  get stage() {
    return this.actor.stage;
  }

  willDestroy() {
    super.willDestroy();

    this._actor = null;
  }

  //  modifier, propertyPath, name, newValue, oldValue

  /**
   * Se produit lorsque n'importe quelle propriété ou additionnelle de l'acteur, d'un accessoire de l'acteur voit sa valeur changer.
   * @method Accessory.BaseAccessory#onActorPropertyChanged
   * @param {Objet} args Arguments.
   * @param {'css'|'render'|'layout'|'other'} args.modifier Type de modification entraînée par le changement.
   * @param {String} args.propertyPath Chemin de la propriété concernée.
   * @param {String} args.name Nom de la propriété concernée.
   * @param {any} args.oldValue Valeur avant changement.
   * @param {any} args.newValue Valeur après changement.
   * @virtual
   */
  onActorPropertyChanged( /*args*/ ) {}

  /**
   * Se produit lorsque n'importe une propriété ou additionnelle de l'acteur, d'un accessoire de l'acteur voit sa valeur changer.
   * @method Accessory.BaseAccessory#onActorValueChanged
   * @param {Objet} args Arguments.
   * @param {'css'|'render'|'layout'|'other'} args.modifier Type de modification entraînée par le changement.
   * @param {String} args.propertyPath Chemin de la propriété concernée.
   * @param {String} args.name Nom de la propriété concernée.
   * @param {any} args.oldValue Valeur avant changement.
   * @param {any} args.newValue Valeur après changement.
   * @virtual
   */
  onActorValueChanged( /*args*/ ) {}

  /**
   * Se produit lors de la lecture de la valeur d'une source de liaison de l'acteur.
   * @method Accessory.BaseAccessory#onActorBindingReadTransform
   * @param {Objet} args Arguments.
   * @param {any} args.value La valeur lue.
   * @param {Binding.BasicBinding} args.binding La liaison concernée.
   * @param {String} args.propertyPath Chemin de la cible de la liaison.
   * @returns la valeur lue.
   * @virtual
   */
  onActorBindingReadTransform(args) {
    return args;
  }

  /**
   * Se produit lors de l'écriture d'une valeur dans une source de liaison de l'acteur.
   * @method Accessory.BaseAccessory#onActorBindingWriteTransform
   * @param {Objet} args Arguments.
   * @param {any} args.value La valeur écrite.
   * @param {Binding.BasicBinding} args.binding La liaison concernée.
   * @param {String} args.propertyPath Chemin de la cible de la liaison.
   * @returns la valeur écrite.
   * @virtual
   */
  onActorBindingWriteTransform(args) {
    return args;
  }

  /**
   * Se produit lors de la destruction de l'acteur.
   * @method Accessory.BaseAccessory#onActorDestroy
   * @virtual
   */
  onActorDestroy() {}

  /**
   * Se produit lors de l'initialisation de l'acteur.
   * @method Accessory.BaseAccessory#onActorInit
   * @virtual
   */
  onActorInit() {}

  /**
   * Se produit juste avant le premier rendu de l'acteur mais après que tous les acteurs de la scène soient initialisés.
   * @method Accessory.BaseAccessory#onActorPostInit
   * @virtual
   */
  onActorPostInit() {}

  /**
   * Se produit à chaque rendu de l'acteur.
   * @method Accessory.BaseAccessory#onActorRender
   * @virtual
   */
  onActorRender() {}

  /**
   * Se produit à chaque fois que l'acteur calcule son style `CSS`.
   * @method Accessory.BaseAccessory#onActorDoComputeStyle
   * @param {Objet} args Paramètres
   * @param {Object} args.style Object de style `CSS`.
   * @virtual
   */
  onActorDoComputeStyle( /*{ style }*/ ) {}

  /**
   * Se produit à chaque fois que l'acteur recalcule son gabarit en cascade.
   * @method Accessory.BaseAccessory#onActorDoLayoutResize
   * @virtual
   */
  onActorDoLayoutResize() {}

  /**
   * Se produit à chaque fois que l'acteur calcule son gabarit.
   * @method Accessory.BaseAccessory#onActorComputeLayout
   * @virtual
   */
  onActorComputeLayout() {}

  /**
   * Se produit à chaque clic souris sur l'acteur.
   * @method Accessory.BaseAccessory#onActorClick
   * @param {Object} args Paramètre.
   * @param {MouseEvent} args.mouseEvent Évènement souris
   * @virtual
   */
  onActorClick( /* { mouseEvent }*/ ) {}

  /**
   * Se produit à chaque entrée de souris sur l'acteur.
   * @method Accessory.BaseAccessory#onActorMouseEnter
   * @param {Object} args Paramètre.
   * @param {MouseEvent} args.mouseEvent Évènement souris
   * @virtual
   */
  onActorMouseEnter( /* { mouseEvent }*/ ) {}


  /**
   * Se produit à chaque sortie de souris sur l'acteur.
   * @method Accessory.BaseAccessory#onActorMouseLeave
   * @param {Object} args Paramètre.
   * @param {MouseEvent} args.mouseEvent Évènement souris
   * @virtual
   */
  onActorMouseLeave( /*{ mouseEvent }*/ ) {}

  /**
   * Se produit à chaque mouvement de souris sur l'acteur.
   * @method Accessory.BaseAccessory#onActorMouseMove
   * @param {Object} args Paramètre.
   * @param {MouseEvent} args.mouseEvent Évènement souris
   * @virtual
   */
  onActorMouseMove( /*{ mouseEvent }*/ ) {}

  /**
   * Se produit à chaque relâchement de bouton de souris sur l'acteur.
   * @method Accessory.BaseAccessory#onActorMouseUp
   * @param {Object} args Paramètre.
   * @param {MouseEvent} args.mouseEvent Évènement souris
   * @virtual
   */
  onActorMouseUp( /*{ mouseEvent }*/ ) {}

  /**
   * Se produit à chaque enfoncement de bouton de souris sur l'acteur.
   * @method Accessory.BaseAccessory#onActorMouseDown
   * @param {Object} args Paramètre.
   * @param {MouseEvent} args.mouseEvent Évènement souris
   * @virtual
   */
  onActorMouseDown( /*{ mouseEvent }*/ ) {}

}

BaseAccessory.registerInAccessoryDefs({ type: 'basic', prototypeSetup: {} });
