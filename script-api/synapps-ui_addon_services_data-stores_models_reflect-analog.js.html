import { isEmpty, isNone } from '@ember/utils';
import { tracked } from '@glimmer/tracking';
import logger from 'logger';
import reflectAnalIcon from 'synapps-ui/assets/reflects/reflectAnal';
import reflectAnalCmdIcon from 'synapps-ui/assets/reflects/reflectAnalCmd';
import ReflectBase, { ReflectBaseSerializer, ReflectTypes } from './reflect-base';

class ReflectAnalogSerializer extends ReflectBaseSerializer {

  deserialize(model, data) {
    super.deserialize(model, data);

    if (data.type === 'analog_thresholds') {
      model.withWitness = true;
    }

    if (data.type === 'analog_Forced') {
      model.isForcible = true;
      model.withWitness = true;
    }

    if (!isEmpty(data.schema.states.value.unit)) {
      model.unit = data.schema.states.value.unit;
    }

    if (!isNone(data.schema.states.value.decimals)) {
      model.decimals = data.schema.states.value.decimals
    }

    if (!isNone(data.schema.states.value.min)) {
      model.min = data.schema.states.value.min
    }

    if (!isNone(data.schema.states.value.max)) {
      model.max = data.schema.states.value.max
    }

    if (model.min === model.max && model.max === 0) {
      model.min = null;
      model.max = null;
    }

    data.schema.commands = data.schema.commands || {};
    data.schema.commands.valueCmd = data.schema.commands.valueCmd || {
      type: data.schema.states.value.type,
      isLinked: false,
    };

    model.isLinked = Boolean(data.schema.commands.valueCmd.isLinked);

    return model;
  }

  deserializeValue(model, data) {
    model.value = data.value;
    model.valueDate = data.valueDate;

    if (model.withWitness) {
      model.witnessMin = data.wSeuilMin;
      model.witnessMax = data.wSeuilMax;
    }

    if (model.isForcible) {
      model.forcedValue = data.valueForced;
    }

    super.deserializeValue(model, data);
    return model;
  }

  serializeCommand(model, { type, valueCmd, valueForcedCmd, forcedCmd, maxValueCmd, minValueCmd }) {
    const args = {
      type,
    };

    if (model._isOldSchema) {
      if (model.isForcible) {
        args.WPTF_ValueForcedCmd = valueForcedCmd;
        args.WPTF_ModeForcedCmd = forcedCmd;
      }

      if (model.allowCommand) {
        args.WPTF_ValueCmd = valueCmd;
      }

      if (model.withWitness) {
        args.WPTF_WSeuilMinCmd = minValueCmd;
        args.WPTF_WSeuilMaxCmd = maxValueCmd;
      }

      return args;
    }

    if (model.isForcible) {
      args.valueForcedCmd = valueForcedCmd;
      args.modeForcedCmd = forcedCmd;
    }

    if (!model.isLinked) {
      args.valueCmd = valueCmd;
    }

    if (model.withWitness) {
      args.wSeuilMinCmd = minValueCmd;
      args.wSeuilMaxCmd = maxValueCmd;
    }

    return args;
  }
}

/**
 * Reflect Analogique.
 * @class Synapps.Reflect.ReflectAnalog
 * @extends Synapps.Reflect.ReflectBase
 * @see Synapps.Reflect.ReflectBase

export default class ReflectAnalog extends ReflectBase {

  static serializer = new ReflectAnalogSerializer();

  static icon = reflectAnalIcon;
  static iconCmd = reflectAnalCmdIcon;

  static type = 'analog';

  _setupStateText() {
    this.stateText = `${this.value}${this.unit}`;
  }

  /**
   * Obtient l'objet de commande du reflet.
   * @member Synapps.Reflect.ReflectAnalog#command
   * @type {Synapps.Reflect.ReflectAnalogCommand}
   * @readonly
   */

  _setupCommand() {

    const setup = {
      type: this.redyType,
      minValueCmd: this.witnessMin,
      maxValueCmd: this.witnessMax,
      valueForcedCmd: this.forcedValue,
      forcedCmd: this.isForced,
      valueCmd: this.value,
    };

    /**
     * Commande de [reflect Analogique]{@link Synapps.Reflect.ReflectAnalog}.
     * @class Synapps.Reflect.ReflectAnalogCommand
     */
    const cmd = new(class {
      /**
       * @member Synapps.Reflect.ReflectAnalogCommand#type
       * @readonly
       * @type {String}
       */
      type = this.redyType;

      /**
       * @member Synapps.Reflect.ReflectAnalogCommand#minValueCmd
       * @type {number}
       */
      @tracked
      minValueCmd = this.witnessMin;

      /**
       * @member Synapps.Reflect.ReflectAnalogCommand#maxValueCmd
       * @type {number}
       */
      @tracked
      maxValueCmd = this.witnessMax;


      /**
       * @member Synapps.Reflect.ReflectAnalogCommand#valueForcedCmd
       * @type {number}
       */
      @tracked
      valueForcedCmd = this.forcedValue;

      /**
       * @member Synapps.Reflect.ReflectAnalogCommand#forcedCmd
       * @type {number}
       */
      @tracked
      forcedCmd = this.isForced;

      /**
       * @member Synapps.Reflect.ReflectAnalogCommand#valueCmd
       * @type {number}
       */
      @tracked
      valueCmd = this.value

      constructor({
        type,
        minValueCmd,
        maxValueCmd,
        valueForcedCmd,
        forcedCmd,
        valueCmd,
      }) {
        this.type = type;

        this.minValueCmd = minValueCmd;
        this.maxValueCmd = maxValueCmd;

        this.valueForcedCmd = valueForcedCmd;
        this.forcedCmd = forcedCmd;

        this.valueCmd = valueCmd
      }
    })(this.pendingCommand ? this.pendingCommand.command : setup);
    return cmd;
  }

  setupCanSubmit() {
    let canSubmit = false;
    if (!this.pendingCommand) {
      if (this.isForcible) {
        canSubmit = this.command.forcedCmd !== this.isForced
          || this.command.forcedCmd && this.command.valueForcedCmd !== this.forcedValue
          || !this.command.forcedCmd && this.command.valueCmd !== this.value;
      } else if (!this.isLinked) {
        canSubmit = this.command.valueCmd !== this.value;
      }

      if (this.withWitness) {
        canSubmit = canSubmit
          || this.command.minValueCmd !== this.witnessMin
          || this.command.maxValueCmd !== this.witnessMax;
      }
    }

    logger.debug(this.name, 'setupCanSubmit, canSubmit', canSubmit);
    this._canSubmit = canSubmit;
  }

  /**
   * @member Synapps.Reflect.ReflectAnalog#hasLimits
   * @readonly
   * @type {number}
   */
  get hasLimits() {
    return !isNone(this.min) && !isNone(this.max);
  }

  /**
   * @member Synapps.Reflect.ReflectAnalog#min

  min;

  /**
   * @member Synapps.Reflect.ReflectAnalog#max

   */
  max;

  /**
   * @member Synapps.Reflect.ReflectAnalog#decimals

   * @type {number}
   */
  decimals;

  /**
   * @member Synapps.Reflect.ReflectAnalog#unit

   */
  unit = '';

  /**
   * @member Synapps.Reflect.ReflectAnalog#isLinked
   * @readonly
   * @type {Boolean}
   */
  isLinked = false;

  /**
   * @member Synapps.Reflect.ReflectAnalog#witnessMin

  witnessMin;

  /**

  forcedValue;

  @tracked
  value;

  /**
   * @member Synapps.Reflect.ReflectAnalog#withWitness
   * @readonly
   * @type {Boolean}
   */
  withWitness = false;

}
