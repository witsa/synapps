import { notifyPropertyChange } from '@ember/object';
import { cancel, debounce as emDebounce, later } from '@ember/runloop';
import logger from 'logger';
import BaseActor from 'synapps-ui/components/actors/base-actor';
import toolsDebounce from 'synapps-ui/tools/debounce';
import isValuesEqual from 'synapps-ui/tools/is-values-equal';
import { inject as service } from '@ember/service';
const STORE_DATA_DELAY = 10;

/**
 * Classe de base pour fournisseur de donnée. Permet de paramétrer les requêtes et d'exposer une donnée pour des liaisons ou accessible par script.
 *
 * @class Actor.DataSource
 * @extends Actor.BaseActor
 */

export default class DataSource extends BaseActor {

  _runner = null;

  @service
  router;

  _autoRefreshing = false;
  get autoRefreshing() {
    return this._autoRefreshing;
  }

  _isFirstRequestDone = false;
  get isFirstRequestDone() {
    return this._isFirstRequestDone;
  }

  _datasourceCanWrite = false;

  // TODO: factoriser lors du dév des reflets ou d'un fournisseur pour les journaux.

  _data = null;

  /**
   * Donnée exposée.
   * @attribute data
   * @type {any}
   */
  get data() {
    return this._data;
  }
  set data(newValue) {
    const oldValue = this._data;
    if (oldValue === newValue) return newValue;
    this._data = this._onWillWriteInData({ oldValue, newValue });
    if (this.properties.writeOnChange) {
      // attention : asynchrone
      this.write();
    }

    notifyPropertyChange(this, 'data');
  }

  _storeData(newValue) {
    if (isValuesEqual(this._data, newValue)) return;
    this._data = newValue;
    notifyPropertyChange(this, 'data');
  }

  _storeDataDebounced(newValue) {
    emDebounce(this, '_storeData', newValue, STORE_DATA_DELAY);
  }

  _isRequesting = false;

  /**
   * Requête en cours ?
   * @attribute isRequesting
   * @type {Boolean}
   * @default false
   */
  get isRequesting() {
    return this._isRequesting;
  }

  set isRequesting(value) {
    this._isRequesting = value;
    notifyPropertyChange(this, 'isRequesting');
  }

  _onInit() {
    this.stage.dataSourcesRegister.record({ record: this });
    super._onInit();
  }

  _canStartAutoRefresh() {
    return this.properties.autoRefresh && this.properties.mode === 'auto';
  }

  _canRequestAtStart() {
    return this.properties.mode === 'auto';
  }

  async _autostart() {
    logger.debug('_autostart()', this.uniqPath, this.properties.mode, 'autoRefresh:', this.properties.autoRefresh);
    if (this._canStartAutoRefresh()) {
      await this._startAutoRefresh();
      return;
    }

    if (this._canRequestAtStart()) {
      await this.requestDebounce();
      return;
    }
  }

  async startAutoRefresh() {
    logger.debug('startAutoRefresh()', this.uniqPath, 'autoRefresh:', this.properties.autoRefresh);
    if (this.properties.autoRefresh) {
      await this._startAutoRefresh();
      return;
    }
  }

  _onAfterPropertyChanged({ propertyPath, ...others }) {
    super._onAfterPropertyChanged({ propertyPath, ...others });

    if (propertyPath === 'properties.autoRefresh') {
      if (this.properties.autoRefresh) {
        this._startAutoRefresh();
      } else {
        this._stopAutoRefresh();
      }
    }
  }

  async _resolve() {
    throw '[DataSource][_resolve] MUST BE IMPLEMENTED!';
  }

  async resolve() {
    await this._resolve();
    return this.data;
  }

  async _startAutoRefresh() {
    logger.debug('_startAutoRefresh', this.uniqPath);
    // await this.resolve();
    if (!this.autoRefreshing) {
      this._autoRefreshing = true;
      return await this._autoRefresh();
    } else {
      return await this.request();
    }
  }

  async _autoRefresh() {
    if (this.isDestroyed || this.isDestroying) return;
    if (this.router.currentRouteName.includes('.login')) {
      this._stopAutoRefresh();
      return;
    }

    await this.request();
    this._runner = later(() => {
      if (this.isDestroying || this.isDestroyed) return;
      if (!this.properties.autoRefresh) return;
      this._autoRefresh();
    }, this.properties.autoRefreshDelay * 1000);
  }

  _stopAutoRefresh() {
    logger.debug('_stopAutoRefresh', this.uniqPath);
    this._autoRefreshing = false;
    if (this._runner) {
      cancel(this._runner);
    }

    this._runner = null;
  }

  /**
   * Effectue une requête retardée. Tant que la méthode est répétée en moins de 300ms, elle ne partira pas.
   * @method requestDebounce
   */
  requestDebounce = toolsDebounce(async function() {
    logger.debug('[DataSource][requestDebounce]');
    return await this.request();
  }, 300);

  /**
   * Effectue une requête.
   * @method request
   * @return {any} La donnée.
   * @async
   */
  async request() {
    if (this.isDestroying || this.isDestroyed) return;
    try {
      this.isRequesting = true;
      let data = await this._request();

      this._storeData(data);
      /**
       * Se produit lorsque la requête s'est terminée.
       * Dans le contexte, on retrouve la donnée dans le champ `data` ou une erreur dans le champ `error`.
       * @event onRequestDone
       * @param {Actor.Event.BaseContext} context
       */
      this.runEvent('onRequestDone', { error: null, data: data })
    } catch (e) {
      logger.error(e);
      this._storeData(null);
      this.runEvent('onRequestDone', { error: e })
    }

    this.isRequesting = false;
    if (!this.isFirstRequestDone) {
      this._isFirstRequestDone = true;
    }

    return this.data;
  }

  /**
   * Effectue une requête d'écriture.
   * @method write
   * @return {any} La donnée.
   * @async
   */
  async write() {
    if (!this._datasourceCanWrite) return this.data;
    let data = null;
    try {
      this.isRequesting = true;
      data = await this._write();

      /**
       * Se produit lorsque la requête d'écriture s'est terminée.
       * Dans le contexte, on retrouve la donnée dans le champ `data` ou une erreur dans le champ `error`.
       * @event onWriteDone
       * @param {Actor.Event.BaseContext} context
       */
      this.runEvent('onWriteDone', { error: null, data: data });
    } catch (e) {
      logger.error(e);
      this.runEvent('onWriteDone', { error: e });
    }

    this.isRequesting = false;
    return data;
  }

  async _request() {
    return null;
  }

  async _write() {
    return null;
  }

  _onWillWriteInData( /*{ oldValue, newValue }*/ ) {
    throw '[DataSource][_onWillWriteInData] NOT IMPLEMENTED!';
  }

  willDestroy() {
    if (this.isDestroyed) return;

    this._stopAutoRefresh();

    this._data = null;

    if (this.scene && this.stage.dataSourcesRegister) {
      this.stage.dataSourcesRegister.unrecord({ record: this });
    }

    super.willDestroy();
  }

  _doComputeStyle({ isAlreadyWrite }) {
    if (isAlreadyWrite)
      return {
        width: 0,
        height: 0
      };

    return {};
  }

  _disposeInParent() {
    // rient à faire;
  }

}
