import { assert } from '@ember/debug';
import logger from 'logger';
import BasicBinding from 'synapps-ui/bindings/basic';
import internalSourceDefs from 'synapps-ui/bindings/internal-source-defs';

/**
 * @class
 * @memberof Binding
 * @extends Binding.BasicBinding
 */
class InternalBinding extends BasicBinding {

  didChangeDataContext({ oldValue, newValue }) {
    // logger.debug("didChangeDataContext", this.target.uniqPath, oldValue, newValue);
    if (oldValue && oldValue.inBindingsRegister) {
      // on enlève la reference du binding.
      // assert('[InternalBinding][didChangeDataContext] oldValue must have inBindingsRegister field.', oldValue.inBindingsRegister);
      oldValue.inBindingsRegister.unrecordBinding(this);
    }

    if (newValue && newValue.inBindingsRegister) {
      // on ajoute la reference du binding.
      // assert('[InternalBinding][didChangeDataContext] newValue must have inBindingsRegister field.', newValue.inBindingsRegister);
      newValue.inBindingsRegister.recordBinding(this);
    }
  }

  static internalSourceDefs = internalSourceDefs;
  static registerInInternalSourceDefs({
    sourceType,
    platform = 'native',
    resolveSource,
    getBasePath,
  }) {
    assert('[InternalBinding][registerInInternalSourceDefs] sourceType is required.', sourceType);
    const internalSourceDef = {
      sourceType,
      platform,
      resolveSource,
      getBasePath,
    };

    this.internalSourceDefs.set(sourceType, internalSourceDef);
  }

  static resolveInternalSource({ sourceType, target, ...others }) {
    assert('[InternalBinding][resolveInternalSource] target is required.', target);
    assert('[InternalBinding][resolveInternalSource] sourceType is required.', sourceType);

    if (!this.internalSourceDefs.has(sourceType)) {
      logger.warn(`[InternalBinding][resolveInternalSource] sourceType [${sourceType}] is unknown.`, this.internalSourceDefs.has(sourceType));
      return null;
    }

    const internalSourceDef = this.internalSourceDefs.get(sourceType);

    return internalSourceDef.resolveSource({ sourceType, target, ...others });
  }

  static getBasePath({ sourceType, ...others }) {
    assert('[InternalBinding][getBasePath] sourceType is required.', sourceType);

    if (!this.internalSourceDefs.has(sourceType)) {
      logger.warn(`[InternalBinding][getBasePath] sourceType [${sourceType}] is unknown.`, this.internalSourceDefs.has(sourceType));
      return null;
    }

    const internalSourceDef = this.internalSourceDefs.get(sourceType);

    return internalSourceDef.getBasePath ? internalSourceDef.getBasePath({ sourceType, ...others }) : null;
  }
}

export default InternalBinding;
