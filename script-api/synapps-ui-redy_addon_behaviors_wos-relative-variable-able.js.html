import { notifyPropertyChange } from '@ember/object';
import { debounce as emDebounce } from '@ember/runloop';
import { isEmpty, isNone } from '@ember/utils';
import { tracked } from '@glimmer/tracking';
import logger from 'logger';
import WosVariableSource from 'synapps-ui-redy/components/actors/wos/wos-variable-source';
import toolsDebounce from 'synapps-ui/tools/debounce';
import isValuesEqual from 'synapps-ui/tools/is-values-equal';

const STORE_DATA_DELAY = 10;

/**
 * @namespace REDY.Behavior
 */

/**
 * Comportement de fournisseur de variable relativement à un autre fournisseur.
 * @class REDY.Behavior.WosRelativeVariableAble
 */

export default function(Class) {
  return class WosRelativeVariableAble extends Class {

    get isAWosVariableSource() {
      return true;
    }

    /**
     * Obtient le fournisseur parent.
     * @member REDY.Behavior.WosRelativeVariableAble#parentSource
     * @type {REDY.Actor.WosVariableSource|REDY.Actor.ResourceSource|REDY.Actor.WosRelativeVariableSource|null}
     * @readonly
     */
    parentSource;

    _datasourceCanWrite = true;

    _relatives = {};

    _isValorized = false;

    _relatives = {};

    _isValorized = false;

    get _toIncludeInRequest() {
      return this.properties.dataReadMode === 'always'
        || this.properties.dataReadMode === 'onetime' && !this._isValorized;
    }

    /**
     * Obtient le chemin complet vers la variable.
     * @member REDY.Behavior.WosRelativeVariableAble#fullPath
     * @type {string}
     * @readonly
     */
    get fullPath() {
      const parentPath = (this.parentSource ? this.parentSource.fullPath : '');
      const separator = (!this.parentSource || isEmpty(this.properties.relativePath) || this.properties.relativePath.startsWith(':')) ? '' : '.';
      return `${parentPath}${separator}${this.properties.relativePath || ''}`;
    }

    /**
     * Obtient le fournisseur de donnée racine.
     * @member REDY.Behavior.WosRelativeVariableAble#dataSource
     * @type {REDY.Actor.WosVariableSource|null}
     * @readonly
     */
    get dataSource() {
      if (this.parentSource && this.parentSource instanceof WosVariableSource) return this.parentSource;
      if (this.parentSource) return this.parentSource.dataSource;
      return null;
    }

    /**
     * Obtient l'attribut à fournir.
     * @member REDY.Behavior.WosRelativeVariableAble#attributeName
     * @type {string|null}
     * @readonly
     */
    get attributeName() {
      let path = this.properties.relativePath || '';
      const index = path.indexOf(':');
      if (index !== -1) {
        path = path.substring(index + 1);
        return isEmpty(path) ? null : path;
      }

      return null;
    }

    /**
     * Obtient le chemin sans l'attribut de la variable à fournir.
     * @member REDY.Behavior.WosRelativeVariableAble#pathPart
     * @type {string|null}
     * @readonly
     */
    get pathPart() {
      let path = this.properties.relativePath || '';
      const index = path.indexOf(':');
      if (index !== -1) {
        path = path.substring(0, index);
      }

      return isNone(path) ? '' : path;
    }

    /**
     * Obtient le chemin complet sans l'attribut de la variable à fournir.
     * @member REDY.Behavior.WosRelativeVariableAble#fullPathVariable
     * @type {string|null}
     * @readonly
     */
    get fullPathVariable() {
      const parentPath = (this.parentSource ? this.parentSource.fullPath : '');
      let pathPart = this.pathPart;
      let separator = '.'
      if (isEmpty(pathPart)) {
        separator = ''
      }

      return `${parentPath}${separator}${pathPart}`;
    }

    @tracked
    _data = null;

    /**
     * Obtient ou définit la donnée exposée.
     * @member REDY.Behavior.WosRelativeVariableAble#data
     * @type {any}
     */
    get data() {
      return this._data;
    }

    set data(newValue) {
      const oldValue = this._data;
      if (oldValue === newValue) return;
      this._data = this._onWillWriteInData({ oldValue, newValue });
      if (this.properties.writeOnChange) {
        // attention : asynchrone
        this.write();
      }

      notifyPropertyChange(this, 'data');
    }

    _runDataEvent(name, args) {
      this.runEvent(name, args);
    }

    _storeData(newValue) {
      if (this.isDestroyed || this.isDestroying) return;
      if (isValuesEqual(this._data, newValue)) return;
      this._data = newValue;
      this._isValorized = true;
      notifyPropertyChange(this, 'data');

      /**
       * Se produit lorsque la donnée est stockée dans [data]{@link REDY.Behavior.WosRelativeVariableAble#data}.
       * @event onDidDataStore
       * @memberof REDY.Behavior.WosRelativeVariableAble#
       * @param {Event.BaseContext} context Le contexte de l'évènement.
       * @param {any} context.data La donnée stockée.
       */
      this._runDataEvent(`onDidDataStore`, { data: newValue });
    }

    _storeDataDebounced(newValue) {
      emDebounce(this, '_storeData', newValue, STORE_DATA_DELAY);
    }

    async _valorize({ isDry = false }) {
      if (!this.dataSource) return;
      try {
        this._storeDataDebounced(await this.dataSource._resolveData({ path: this.fullPathVariable, attributeName: this.attributeName, isDry }));
      } catch (e) {
        logger.warn(`[WosRelativeVariableAble][_valorize] Error during resolve relative ${this.uniqPath} ${this.fullPath}`);
      }
    }

    async _valorizeRelatives({ isDry = false }) {
      if (!this.dataSource) return;

      await this._valorize({ isDry });

      const relatives = Object.values(this._relatives);
      for (const relative of relatives) {
        if (relative.isDestroyed || relative.isDestroying) continue;
        await relative._valorize({ isDry });
      }
    }

    _onWillWriteInData({ /*oldValue,*/ newValue }) {
      if (!this.dataSource) return newValue;
      if (isEmpty(this.fullPathVariable)) return newValue;
      if (this.attributeName !== 'value') return newValue;
      const variable = this.dataSource.dataStores.redy.getWosVariable(this.fullPathVariable);
      if (variable) {
        variable.value = newValue;
      }

      return newValue;
    }

    _onEndingRender({ isFirstRender }) {
      if (this.isDestroyed || this.isDestroying) return;
      if (isFirstRender) {
        this._setupParentSource();
      }
    }

    async _write() {
      if (!this.dataSource) return;
      if (isEmpty(this.fullPathVariable)) return;
      if (this.attributeName !== 'value') return;
      const variable = await this.dataSource.dataStores.redy.writeWosVariableValue(this.fullPathVariable);
      return await this.dataSource._resolveData({ path: variable.path, attributeName: this.attributeName });
    }

    /**
     * Effectue une requête d'écriture.
     * @method write
     * @memberof REDY.Behavior.WosRelativeVariableAble#
     * @return {any} La donnée.
     * @async
     */
    async write() {
      if (!this._datasourceCanWrite) return this.data;
      let data = null;
      try {
        this.isRequesting = true;
        data = await this._write();
        /**
         * Se produit lorsque la requête d'écriture s'est terminée.
         * @event onWriteDone
         * @memberof REDY.Behavior.WosRelativeVariableAble#
         * @param {Event.BaseContext} context
         * @param {String|Object|null} context.error Une erreur éventuelle
         * @param {any} context.data La donnée
         */
        this._runDataEvent(`onWriteDone`, { error: null, data: data });
      } catch (e) {
        logger.error(e);
        this._runDataEvent(`onWriteDone`, { error: e });
      }

      this.isRequesting = false;
      return data;
    }

    _invalorize() {
      this._isValorized = false;
      Object.values(this._relatives).forEach(relative => relative._invalorize());
    }

    _doAfterPropertyChanged({ name, ...others }) {
      if (this.isDestroying || this.isDestroyed) return;
      super._doAfterPropertyChanged({ name, ...others });

      if (name === 'relativeTo') {
        this._invalorize();
        this._setupParentSource();
        this._valorizeOrRequest();

        return;
      }

      if (name === 'relativePath') {
        this._invalorize();
        this._valorizeOrRequest();

        return;
      }
    }

    _valorizeOrRequest = toolsDebounce(async () => {
      if (this.dataSource) {
        if (['auto', 'relative'].includes(this.dataSource.properties.mode) && this.scene.stageIsReady) {
          this.dataSource.requestDebounce();
        } else {
          await this._valorizeRelatives({ isDry: true });
        }
      }
    }, 150);

    _setupParentSource() {
      this._detachParentSource();
      let relativeTo = this.properties.relativeTo;
      if (isEmpty(relativeTo)) {
        this._attachParentSource(this.dataContext);
      } else {
        let stage = this.stage;
        const words = relativeTo.split('/');
        if (words.length > 1 && words[0] === 'global') {
          relativeTo = words[1];
          stage = this.scene.synapp.backstage;
        }

        const parentSource = stage.getActor(relativeTo);
        this._attachParentSource(parentSource);
      }
    }

    _attachParentSource(parentSource) {
      if (!parentSource || !parentSource.isAWosVariableSource) return;
      if (this.parentSource === parentSource) return;
      this.parentSource = parentSource;
      this.parentSource._relatives[this.uniqPath] = this;
      if (!this.dataSource || this.stage.dataSourcesRegister.hasKey(this.dataSource.key)) return;
      if (this.stage.dataSourcesRegister.hasRecord({ record: this.dataSource })) return;
      this.stage.dataSourcesRegister.record({ record: this.dataSource });
    }

    _detachParentSource(parentSource) {
      this._invalorize();
      parentSource = parentSource || this.parentSource;
      if (!parentSource) return;
      delete parentSource._relatives[this.uniqPath];

      if (
        this.dataSource
        && this.dataSource.stage.args.isBackstage
        && this.dataSource.properties.autoRefresh
        && this.dataSource.properties.mode === 'relative'
        && Object.keys(this.dataSource._relatives).length === 0
      ) {
        this.dataSource._stopAutoRefresh();
      }

      this.parentSource = null;
    }

    didChangeDataContext({ oldValue, newValue }) {
      if (!isEmpty(this.properties.relativeTo)) return;
      if (oldValue === newValue) return;
      if (oldValue && oldValue.isAWosVariableSource) {
        // on enlève la reference.
        this._detachParentSource(oldValue);
      }

      if (newValue && newValue.isAWosVariableSource) {
        // on ajoute la reference du binding.
        this._setupParentSource();
        this._valorizeOrRequest();
      }
    }

    willDestroy() {
      if (this.isDestroyed) return;
      this._relatives = {};
      this._detachParentSource();
      super.willDestroy();
    }

  };
}
