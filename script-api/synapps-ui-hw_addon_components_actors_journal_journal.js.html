const REQUEST_TRIGGER_FIELDS = [
  'properties.mode',
  'properties.pageSize',
  'properties.categories',

const AUTO_REFRESH_TRIGGER_FIELDS = [
  'properties.dateTo',
];

export default class Journal extends BaseRequestList {

  // Need to remove occurrences column when actor property mode is not news
  // EV? some duplicated code from BaseRequestList: how to avoid this ?
  @computed('properties.{columns,prototypeSetup.columns.options,mode}')
  get columns() {
    try {
      const columns = JSON.parse(this.properties.columns);
      const cols = isEmpty(columns) ? [...this.properties.prototypeSetup.columns.options] : columns;
      return this.properties.mode === 'news' ? cols : cols.filter(c => c !== 'occurrences');
    } catch (e) {
      logger.warn('[MultiSelectRequestList][get columns] JSON parsing error.', this.properties.columns);
      return [...this.properties.prototypeSetup.columns.options];
    }
  }

  /**
   * Magasin de donnée Highway.
   * @member HW.Actor.Journal#highway
   * @type {HW.Synapps.HighwayStore}
   * @readonly
   */
  @service('synapps-ui-hw/highway-store')
  highwayStore;

  //BEHAVIOUR = BEHAVIOUR;
  TYPE_ICONS = BehaviourIcons;
  TYPE_TEXTS = BehaviourTexts;
  COLUMN_TEMPLATES = COLUMN_TEMPLATES;
  REQUEST_TRIGGER_FIELDS = REQUEST_TRIGGER_FIELDS;
  AUTO_REFRESH_TRIGGER_FIELDS = AUTO_REFRESH_TRIGGER_FIELDS;

  @tracked
  _isAcking = false;

  @tracked
  _ackComment;

  @tracked
  _selectedEvent = null;

  @computed('_selectedEvent')
  get selectedEvent() {
    return this._selectedEvent;
  }

  set selectedEvent(value) {
    if (this._selectedEvent === value) return;
    this._selectedEvent = value;

    notifyPropertyChange(this, 'selectedEvent');
    if (this.isDestroyed || this.isDestroying) return;
    this.runEvent('onSelectChange', { value });
  }

  @computed('properties.dateFrom')
  get dateFrom() {
    return this.properties.dateFrom ? moment(this.properties.dateFrom) : null;
  }

  @computed('properties.dateTo')
  get dateTo() {
    return this.properties.dateTo ? moment(this.properties.dateTo) : null;
  }

  @readOnly('rows')
  journalEvents;

  _canAutoRefreshFx() {
    return super._canAutoRefreshFx()
      && (!this.to || this.to.isSameOrAfter(moment()))
      && this.page === 1
      && !this._isAcking;
  }

  async _request() {
    return await this.loadJournal();
  }

  async loadJournal() {
    const filters = {
      pageSize: this.properties.pageSize,
      pageIndex: this.page
    };

    switch (this.properties.mode) {
      case 'pending':
        filters.mode = 2;
        break;
      case 'news':
        filters.mode = 3;
        break;
      case 'all':
      default:
        filters.mode = 1;
        break;
    }

    if (this.properties.categories) {
      filters.categories = JSON.parse(this.properties.categories);
    }

    if (!isEmpty(this.properties.searchText)) {
      filters.search = this.properties.searchText;
    }

    if (!isNone(this.properties.resourceType)) {
      filters.resourceType = this.properties.resourceType;
    }

    if (this.properties.resources) {
      filters.resourceIds = JSON.parse(this.properties.resources);
    }

    if (this.properties.scopes) {
      filters.scopeIds = JSON.parse(this.properties.scopes);
    }

    if (this.properties.infras) {
      var infraIds = JSON.parse(this.properties.infras);
      var index = infraIds.indexOf(null);
      if (index > -1) {
        infraIds[index] = 0;
      }

      filters.infraIds = infraIds;
      filters.includeInfraChildren = this.properties.includeInfraChildren;
    }

    if (!isNone(this.properties.criticity)) {
      filters.criticity = this.properties.criticity;
    }

    if (this.dateFrom) {
      filters.after = this.properties.dateFrom;
    }

    if (this.dateTo) {
      filters.before = this.properties.dateTo;
    }

    const { count, events } = await this.highwayStore.searchJournal(filters);

    // get or request referenced resources
    await this._loadReferences(events);

    const rows = events;
    return { count, rows };
  }

  async _loadReferences(events) {

    // get or request users
    var usersIds = [];
    events.forEach(evt => {
      if (!isNone(evt.userActionId) && !usersIds.includes(evt.userActionId)) usersIds.push(evt.userActionId);
      if (!isNone(evt.userAckId) && !usersIds.includes(evt.userAckId)) usersIds.push(evt.userAckId);
    })

    if (!isEmpty(usersIds)) await this.highwayStore.getManyUsers(usersIds);

    // get or request reflects classes
    await this.highwayStore.searchReflectClasses();

    // get journal resources. Ex: { flux: [3, 8, 15], reflect: [2, 5, 12] }
    var references = {};

    var resourceRefs = events.flatMap(event => {
      var keys = Object.keys(event.resources);
      return keys.map(k => event.resources[k]);
    });

    resourceRefs.forEach(res => {
      var resourceIds = references[res.type];
      if (isNone(resourceIds)) {
        references[res.type] = [res.id];
      } else if (!resourceIds.includes(res.id)) {
        resourceIds.push(res.id);
      }
    })

    // ensure direct references are loaded
    if (!isNone(references.smartLocalUnit)) {
      await this.highwayStore.getManySmartLocalUnits(references.smartLocalUnit);
    }

    var reflects = isNone(references.reflect) ? [] : await this.highwayStore.getManyReflects(references.reflect);
    var fluxes = isNone(references.flux) ? [] : await this.highwayStore.getManyFluxs(references.flux);

    // ensure indirect smartLocalUnits references (throught reflects and fluxes) are loaded
    await this.highwayStore.getSmartLocalUnitReferences(reflects.concat(fluxes));
  }

  _resetData() {
    super._resetData();
    if (!this._isAcking) {
      this.selectedEvent = null;
    }
  }

  async startAck(event) {
    this._isAcking = true;
    this._ackComment = event.ackComment;
    this._stopAutoRefresh();
    this.runEvent('onStartAcking', { event });
  }

  async ack(eventId, comment) {
    this._isAcking = false;
    var event = await this.highwayStore.ackJournal(eventId, comment);
    this._startOrStopRefresh();
    this.runEvent('onEndAcking', { event });
  }

  async cancelAck() {
    this._isAcking = false;
    this._startOrStopRefresh();
  }

  willDestroy() {
    if (this.isDestroyed) return;
    this._selectedEvent = null;
    super.willDestroy();
  }

  @action
  async activeRow(event, column) {
    this.selectEvent(event);
    if (column.key === 'ack') {
      if (this._isAcking) {
        await this.cancelAck();
      }
      else {
        await this.startAck(event);
      }
    }
  }

  @action
  selectEvent(event) {
    this.selectedEvent = event;
  }

  @action
  async _ack(event, mouseEvent) {
    mouseEvent.stopPropagation();
    await this.ack(event.id, this._ackComment);
  }

  @action
  async _cancelAck(mouseEvent) {
    mouseEvent.stopPropagation();
    await this.cancelAck();
  }
}
