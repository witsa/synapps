  /**
   * Obtient ou charge la liste des scopes (maximum : 200).
   * @member HW.Synapps.HighwayStoreService#searchScopes
   * @param {Object} params Paramètres.
   * @param {Boolean} [params.forceLoad=false] Forcer le chargement de la liste des scopes.
   * @param {Number} [params.pageSize=200] Nombre de scopes à récupérer.
   * @returns {Promise<Array.<Object>>} Liste des scopes.
   * @async
   */
  @authenticatedRequest
  async searchScopes(params = {}) {
    const filters = Object.assign({ forceLoad: false, pageSize: PAGE_SIZE }, params);
    if (!filters.forceLoad && this._scopesRegister.isRequestedYet) {
      return this._scopesRegister.records;
    }

    const results = await this.searchResources({ resourceType: 'scope', filters });

    for (const record of [...results.data]) {
      if (this._scopesRegister.hasRecord({ record })) continue;
      this._deserializeScope(record, true);
    }

    this._scopesRegister._isRequestedYet = true;
    return this._scopesRegister.records;
  }

  findWum(id, type) {
    switch (type) {
      case 'unit':
        return this._wumUnitsRegister.findByKey(id);
      default:
        return null;
    }

  _deserializeScope(record, cache) {
    if (cache) this._scopesRegister.record({ record });
    return record;
  }

  _deserializeInfra(record, cache) {
    var infra = this._infrasRegister.findByRecord({ record });
    if (isNone(infra)) {
      infra = new Infra();
      Infra.serializer.deserialize(infra, record);
      if (cache) this._infrasRegister.record({ record: infra });
    } else {
      Infra.serializer.deserialize(infra, record);
    }

    return infra;
  }

  _deserializeSmartLocalUnit(record, cache) {
    var smartlocalunit = this._smartlocalunitsRegister.findByRecord({ record });
    if (isNone(smartlocalunit)) {
      smartlocalunit = new SmartLocalUnit();
      SmartLocalUnit.serializer.deserialize(smartlocalunit, record);
      if (cache) this._smartlocalunitsRegister.record({ record: smartlocalunit });
    } else {
      SmartLocalUnit.serializer.deserialize(smartlocalunit, record);
    }

    return smartlocalunit;
  }

  _deserializeFlux(record, cache) {
    var flux = this._fluxsRegister.findByRecord({ record });
    if (isNone(flux)) {
      flux = new Flux();
      Flux.serializer.deserialize(flux, record);
      if (cache) this._fluxsRegister.record({ record: flux });
    } else {
      Flux.serializer.deserialize(flux, record);
    }

    return flux;
  }

  _deserializeFluxItem(record, cache) {
    var fluxItem = this._fluxitemsRegister.findByRecord({ record });
    if (isNone(fluxItem)) {
