import { action, computed, notifyPropertyChange, set } from '@ember/object';
import { readOnly } from '@ember/object/computed';
import { isNone } from '@ember/utils';
import { all } from 'rsvp';
import BaseActor from 'synapps-ui/components/actors/base-actor';
import Parenthood from 'synapps-ui/components/behaviors/parenthood';

@Parenthood
class ParenthoodBaseActor extends BaseActor {}

/**
 * Base pour les acteurs de disposition.
 *
 * @class Actor.Layout.BaseLayout
 * @extends Behavior.Parenthood
 * @extends Actor.BaseActor
 * @private
 */
class LayoutBase extends ParenthoodBaseActor {

  /**
   * Obtient les propriétés de l'acteur.
   * @member properties
   * @memberof Actor.Layout.BaseLayout#
   * @type {Actor.Layout.BaseLayoutProperties}
   * @readonly
   */

  /**
   * Obtient les setups des acteur enfants.
   * @member childSetups
   * @memberof Actor.Layout.BaseLayout#
   * @readonly
   * @type {Array.<JSON.ActorSetup>}
   */
  get childSetups() {
    return this.args.setup.children;
  }

  set childSetups(array) {
    // TODO ça plante ça normalement il faut le virer
    this.args.setup.children = array;
  }

  constructor(owner, args) {
    super(owner, args);
    if (!args.setup.children) {
      args.setup.children = [];
    }
  }

  /**
   * Pour savoir s'il est possible d'ajouter un acteur.
   * @member canAddActor
   * @memberof Actor.Layout.BaseLayout#
   * @readonly
   * @type {boolean}
   */
  canAddActor() {
    return !this.adding;
  }

  /**
   * Pour savoir s'il est possible de retirer un acteur.
   * @member canRemoveActor
   * @memberof Actor.Layout.BaseLayout#
   * @readonly
   * @type {boolean}
   */
  canRemoveActor() {
    return this.childSetups.length > 0;
  }

  /**
   * Pour savoir s'il est possible de retirer tous les acteurs.
   * @member canClearActors
   * @memberof Actor.Layout.BaseLayout#
   * @readonly
   * @type {boolean}
   */
  canClearActors() {
    return this.childSetups.length > 0;
  }

  @action
  onAddedChildDidRender(childUi) {
    super.onAddedChildDidRender.bind(super.this)(childUi);
    if (this.adding) {
      this.adding.actor = childUi;
      this._doCascadeRender();
      this.adding.gen.next().value;
      this.adding = null;
    }

    if (this.addings) {
      const adding = this.addings.findBy('key', childUi.key);
      if (!adding) return;
      adding.childUi = childUi;
      adding.gen.next().value;
    }
  }

  /**
   * Déplace vers le haut un acteur enfant obtenu par sa clé.
   * @method moveUpActor
   * @memberof Actor.Layout.BaseLayout#
   * @param {String} key Clé de l'acteur enfant à déplacer.
   */
  moveUpActor(key) {
    const actorSetup = this.childSetups.findBy('key', key);
    if (!actorSetup) return;
    if (this.childSetups.firstObject === actorSetup) return;
    const actorSetupIndex = this.childSetups.indexOf(actorSetup);
    this.childSetups.splice(actorSetupIndex - 1, 2, actorSetup, this.childSetups[actorSetupIndex - 1]);
    notifyPropertyChange(this, 'childSetups');
    this._sendAllSizesToPreviewer();
  }

  /**
   * Déplace vers le bas un acteur enfant obtenu par sa clé.
   * @method moveDownActor
   * @memberof Actor.Layout.BaseLayout#
   * @param {String} key Clé de l'acteur enfant à déplacer.
   */
  moveDownActor(key) {
    const actorSetup = this.childSetups.findBy('key', key);
    if (!actorSetup) return;
    if (this.childSetups.lastObject === actorSetup) return;
    const actorSetupIndex = this.childSetups.indexOf(actorSetup);
    this.childSetups.splice(actorSetupIndex, 2, this.childSetups[actorSetupIndex + 1], actorSetup);
    notifyPropertyChange(this, 'childSetups');
    this._sendAllSizesToPreviewer();
  }

  /**
   * Ajoute un acteur enfant.
   * L'acteur est ajouté à la fin de la liste si `atIndex` n'est pas renseigné.
   * Note: C'est le seul moyen de créer un acteur.
   * @todo donner des exemples ou liens vers guide.
   * @method addActor
   * @memberof Actor.Layout.BaseLayout#
   * @param {JSON.ActorSetup} actorSetup Setup JSON de l'acteur à ajouter.
   * @param {Number} [atIndex] Position à laquelle l'acteur doit être ajouté.
   * @async
   * @returns {Promise<Actor.BaseActor>} l'acteur créé
   */
  async addActor(actorSetup, atIndex) {

    if (!this.canAddActor()) return;

    const adding = {};

    const layout = this;

    adding.promise = new Promise(resolve => {
      const generator = function*() {

        /**
         * Se produit lorsque un acteur est ajouté.
         *
         * @event onDidActorAdd
         * @memberof Actor.Layout.BaseLayout#
         * @param {Event.BaseContext} context Contexte de l'évènement.
         * @param {Actor.BaseActor} context.addedActor Acteur ajouté.
         */
        layout.runEvent('onDidActorAdd', { addedActor: adding.actor });
        layout._sendAllSizesToPreviewer();
        return yield resolve(adding.actor);
      }

      adding.gen = generator();
      if (isNone(atIndex)) {
        this.childSetups.pushObject(actorSetup);
      } else {
        this.childSetups.insertAt(atIndex, actorSetup);
      }
    });

    this.adding = adding;

    return await this.adding.promise;
  }

  /**
   * Ajoute une collection d'acteurs enfants.
   * Les acteurs sont ajoutés à la fin de la liste.
   * @todo donner des exemples ou liens vers guide.
   * @method addActors
   * @memberof Actor.Layout.BaseLayout#
   * @param {Array.<JSON.ActorSetup>} actorSetups Tableau des Setups JSON des acteurs à ajouter.
   * @async
   * @returns {Promise<Array.<Actor.BaseActor>>} le tableau des acteurs créés
   */
  async addActors(actorSetups) {

    if (!this.canAddActor()) return;

    const addings = [];
    const layout = this;

    addings.promise = null;

    for (const actorSetup of actorSetups) {
      const adding = {
        key: actorSetup.key,
      };

      adding.promise = new Promise(resolve => {
        const generator = function*() {
          layout.runEvent('onDidActorAdd', { addedActor: adding.actor });
          return yield resolve(adding.actor);
        }

        adding.gen = generator();
      });

      addings.push(adding);
    }

    addings.promise = all(addings.mapBy('promise'));

    this.childSetups.pushObjects(actorSetups);

    this.addings = addings;

    await this.addings.promise;

    const childUis = this.addings.mapBy('childUi');

    this._doCascadeRender();

    /**
         * Se produit lorsque une collection d'acteurs est ajoutée.
         *
         * @event onDidActorsAdd
         * @memberof Actor.Layout.BaseLayout#
         * @param {Event.BaseContext} context Contexte de l'évènement.
         * @param {Array.<Actor.BaseActor>} context.addedActors Acteurs ajoutés.
         */
    layout.runEvent('onDidActorsAdd', { addedActors: childUis });
    layout._sendAllSizesToPreviewer();

    this.adding = null;

    return childUis;
  }

  /**
   * Retire un acteur enfant.
   * @method removeActor
   * @memberof Actor.Layout.BaseLayout#
   * @param {String} actorKey Clé de l'acteur enfant à retirer.
   */
  removeActor(actorKey) {
    if (!this.canRemoveActor()) return;
    const childSetup = this.childSetups.findBy('key', actorKey);
    if (!childSetup) return;
    this.childSetups.removeObject(childSetup);
    this._sendAllSizesToPreviewer();
  }

  /**
   * Retire  tous les acteurs enfants.
   * @method clearActors
   * @memberof Actor.Layout.BaseLayout#
   */
  clearActors() {
    if (!this.canClearActors()) return;
    this.childSetups.clear();
    set(this, 'childSetups', this.childSetups);
    this._sendAllSizesToPreviewer();
  }

  _doComputeStyle({ properties, style = {}, isAlreadyWrite = false }) {
    super._doComputeStyle({ properties, isAlreadyWrite, style });
    this.styling.setupStyleKey(style, "overflowX", properties, isAlreadyWrite);
    this.styling.setupStyleKey(style, "overflowY", properties, isAlreadyWrite);
    return style;
  }

  // [MIGRATION V1]

  _getEventContextV1Class() {

    const BaseClass = super._getEventContextV1Class();

    return class extends BaseClass {
      @readOnly('_objectUi.isParent')
      withChildren;

      @computed('_objectUi.childUis.size')
      get actors() {
        return [...this._objectUi.childUis].mapBy('_eventContextV1');
      }

      findByKey(key) {
        const childUi = this._objectUi.getActor(key);
        return childUi ? childUi._eventContextV1 : null;
      }

      findByLabel(label) {
        return this.findByKey(label);
      }
    };
  }
}


export default LayoutBase;
