import { readOnly } from '@ember/object/computed';
import { tracked } from '@glimmer/tracking';
import moment from 'moment';
import RedySession from 'synapps-ui-redy/redy-session';
import RedyUser from 'synapps-ui-redy/redy-user';
import BaseHost from 'synapps-ui/bases/host';
import V1EventScript from 'synapps-ui/behaviors/v1-event-script';

// TODO déplacer la gestion de la sauvegarde de session dans la session!!

@V1EventScript
class V1ScriptedHost extends BaseHost {}

/**
 * @namespace REDY.Synapps
 */

/**
 * Représente l'ULI hôte.
 * @class REDY.Synapps.RedyHost
 * @extends Synapps.BaseHost
 */
class RedyHost extends V1ScriptedHost {

  /**
   * La session pour cet hôte.
   * @member session
   * @memberof REDY.Synapps.RedyHost
   * @readonly
   * @type {REDY.Synapps.RedySession}
   */

  /**
   * Obtient le magasin de donnée de REDY.
   * @member store
   * @memberof REDY.Synapps.RedyHost
   * @readonly
   * @type {REDY.Synapps.RedyStore}
   */
  _store;
  get store() {
    return this._store;
  }

  get _localStorage() {
    return this.store._localStorage;
  }

  onAuthenticationError( /*error*/ ) {

  }

  _namespace = 'web/api/v1';
  get namespace() {
    return this._namespace;
  }

  get _localStorageSessionKey() {
    return `synapps_session_REDY_${this._originalDomain || this.domain}`;
  }

  get _oldLocalStorageSessionKey() {
    return `${this._originalDomain ||this.domain}/${this.namespace}_session`;
  }

  get _localStorageSession() {

    let str = this._localStorage.getItem(this._oldLocalStorageSessionKey);
    if (str) {
      this._localStorage.setItem(this._localStorageSessionKey, str);
      this._localStorage.removeItem(this._oldLocalStorageSessionKey);
    }

    str = this._localStorage.getItem(this._localStorageSessionKey);
    return str ? JSON.parse(str) : null;
  }

  set _localStorageSession(session) {
    this._localStorage.setItem(this._localStorageSessionKey, JSON.stringify(session));
  }

  _clearLocalStorageSession() {
    this._localStorage.removeItem(this._localStorageSessionKey);
  }

  _saveLocalStorageSession() {
    const { token, scope, expirationDate, inSid, user, sid, picture } = this.session;
    const basic = this.session._keepUserLogged ? this.session.basic : null;
    this._localStorageSession = {
      inSid,
      basic,
      sid,
      token,
      scope,
      expirationDate,
      picture,
      user: user.toStoreObject(),
    };
  }

  _restoreFromLocalStorageSession() {
    if (this._localStorageSession) {
      this.session._scope = this._localStorageSession.scope;
      this.session._token = this._localStorageSession.token;
      this.session._inSid = this._localStorageSession.inSid;
      this.session._basic = this._localStorageSession.basic;
      this.session._sid = this._localStorageSession.sid;
      this.session._picture = this._localStorageSession.picture;
      this.session._user = new RedyUser(this._localStorageSession.user);
      this.session._expirationDate = moment(this._localStorageSession.expirationDate);
      if (this.session.basic) {
        this.session._keepUserLogged = true;
      }
    }
  }

  constructor() {
    super({ session: new RedySession() });
  }

  /**
   * Nom de l'ULI
   * @member name
   * @memberof REDY.Synapps.RedyHost
   * @readonly
   * @type {string}
   */
  @tracked
  _name;
  @readOnly('_name')
  name;

  /**
   * WID de l'ULI
   * @member wid
   * @memberof REDY.Synapps.RedyHost
   * @readonly
   * @type {string}
   */
  @tracked
  _wid
  @readOnly('_wid')
  wid;

  /**
   * hard de l'ULI
   * @member hard
   * @memberof REDY.Synapps.RedyHost
   * @readonly
   * @type {string}
   */
  @tracked
  _hard;
  @readOnly('_hard')
  hard;

  /**
   * Type d'ULI
   * @member type
   * @memberof REDY.Synapps.RedyHost
   * @readonly
   * @type {string}
   */
  @tracked
  _type;
  @readOnly('_type')
  type;

  /**
   * Version d'ULI
   * @member version
   * @memberof REDY.Synapps.RedyHost
   * @readonly

   */
  @tracked
  _version;
  @readOnly('_version')
  version;

  /**
   * Date de version d'ULI. Format : Date ISO.
   * @member versionDate
   * @memberof REDY.Synapps.RedyHost
   * @readonly
   * @type {string}
   */
  @tracked
  _versionDate;
  @readOnly('_versionDate')
  versionDate;

  /**
   * URL des application RIA qu'utilise l'ULI.

   * @memberof REDY.Synapps.RedyHost

   * @type {string}
   */
  @tracked
  _riaUrl;
  @readOnly('_riaUrl')
  riaUrl;

   * @readonly

   */
  @tracked
  _picture;
  @readOnly('_picture')
  picture;

  _profil = '';

  @tracked
  _optAdds = {};
  @readOnly('_optAdds')
  optAdds;

  // [MIGRATION V1]

  _getEventContextV1Class() {

    const BaseClass = super._getEventContextV1Class();

    return class extends BaseClass {

      @readOnly('_objectUi.domain')
      id;

      @readOnly('_objectUi.name')
      name;

      wid;

      hard;

      versionDate;

      @readOnly('_objectUi.picture')
      picture;

      @readOnly('_objectUi.riaUrl')
      riaUrl;

      @readOnly('_objectUi.session._eventContextV1')
      session;

    };
  }

}
