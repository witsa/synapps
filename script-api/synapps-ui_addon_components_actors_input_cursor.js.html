import { action, computed, notifyPropertyChange } from '@ember/object';
import { later, throttle } from '@ember/runloop';
import { isPresent } from '@ember/utils';
import { tracked } from '@glimmer/tracking';
import BaseActor from 'synapps-ui/components/actors/input/base-input';
import prototypeSetup from 'synapps-ui/components/actors/input/cursor-prototype-setup';
import hasOwnProperty from 'synapps-ui/tools/has-own-property';

const toLowerCase = (v) => {
  return v && v.toLowerCase ? v.toLowerCase() : v;
}

/**
 * Acteur Curseur.
 * Type : **input/cursor**
 *
 * @class Actor.Input.Cursor
 * @extends Actor.Input.BaseInput
 */
export default class Cursor extends BaseActor {

  @tracked
  _isDisplayed = true;


  /**
   * Obtient la valeur pendant le glissement du curseur.
   * @member slidingValue
   * @memberof Actor.Input.Cursor#
   * @type {Number}
   * @readonly
   */
  @tracked
  _slidingValue;
  get slidingValue() {
    return this._slidingValue;
  }
  set slidingValue(value) {
    this._slidingValue = value;
    notifyPropertyChange(this, 'slidingValue');
  }

  /**
   * Obtient la deuxième valeur pendant le glissement du curseur.
   * @member slidingValue2
   * @memberof Actor.Input.Cursor#
   * @type {Number}
   * @readonly
   */
  @tracked
  _slidingValue2;
  get slidingValue2() {
    return this._slidingValue2;
  }
  set slidingValue2(value) {
    this._slidingValue2 = value;
    notifyPropertyChange(this, 'slidingValue2');
  }

  @computed('properties.{isSingleValue,value,value2}')
  get _start() {
    const value = this.properties.value;
    const value2 = this.properties.value2;
    return this.properties.isSingleValue ? [value] : [value, value2];
  }

  @computed('properties.barPosition')
  get _barPosition() {
    // from v1
    return toLowerCase(this.properties.barPosition || 'default');
  }

  @computed('properties.{isSingleValue,barCenter}', '_barPosition')
  get _connect() {
    let value = false;
    switch (this._barPosition) {
      case 'lower':
        value = this.properties.isSingleValue ? [true, false] : [false, true, false];
        break;
      case 'upper':
        value = this.properties.isSingleValue ? [false, true] : [true, false, true];
        break;
      case 'default':
      default:
        break;
    }

    return value;
  }

  @computed('properties.{min,max,step}')
  get _range() {
    const min = this.properties.min;
    var max = this.properties.max;
    var isValid = isPresent(min) && !isNaN(min) && isPresent(max) && !isNaN(max) && (max > min);
    return isValid
      ? {
        'min': this._fixValue(min),
        'max': this._fixValue(max)
      }
      : {
        'min': this._fixValue(0),
        'max': this._fixValue(100)
      }
  }

  @computed('properties.{orientation,withTooltips}')
  get _cursorStyle() {
    return `
  ${this.properties.orientation === 'vertical' ? 'height' : 'width'}: 100%;
  `;
  }

  @computed('properties.step')
  get _step() {
    let value = parseFloat(this.properties.step);
    value = isNaN(value) ? null : value;
    return value;
  }

  @computed('properties.{isSingleValue,limit}', '_step')
  get _limit() {
    if (this.properties.isSingleValue) return undefined;
    let limit = parseFloat(this.properties.limit);
    if (isNaN(limit)) return undefined;
    if (limit < this._step) return this._step;
    return this._fixValue(this.properties.limit);
  }

  @computed('properties.{isSingleValue,margin,step}')
  get _margin() {
    if (this.properties.isSingleValue) return undefined;
    let margin = parseFloat(this.properties.margin);
    if (isNaN(margin)) return undefined;
    if (margin < 0) return 0;
    return this._fixValue(this.properties.margin);
  }

  get _format() {
    return {
      // 'to' the formatted value. Receives a number.
      to: (value) => {
        return this._fixValue(value);
      },
      // 'from' the formatted value.
      // Receives a string, should return a number.
      from: (value) => {
        return this._fixValue(parseFloat(value));
      }
    };
  }

  @computed('properties.enabled')
  get disabled() {
    return !this.properties.enabled;
  }
  set disabled(value) {
    // eslint-disable-next-line ember/no-assignment-of-untracked-properties-used-in-tracking-contexts
    this.properties.enabled = !value;
    notifyPropertyChange(this, 'disabled');
    // eslint-disable-next-line no-setter-return
    return !this.properties.enabled;
  }

  @action
  _didElementInsert(element) {
    this.slidingValue = this.properties.value;
    this.slidingValue2 = this.properties.value2;
    super._didElementInsert.bind(super.this)(element);
  }

  @action
  _updateSlider() {
    this._isDisplayed = false;
    later(() => {
      if (this.isDestroying || this.isDestroyed) return;
      this._isDisplayed = true;
    }, 150);
  }

  /**
   * fix javascript problem about value with decimal. Ex 2.00000000001
   * @method Actor.Input.Cursor#_fixValue
   * @private
   * @ignore
   * @param {*} val
   */
  _fixValue(val) {
    const step = parseFloat(this._step);
    if (isNaN(step)) return val;
    const round = Math.round(val / step) * step;
    return parseFloat(round.toFixed(Math.abs(Math.min(0, Math.floor(Math.log10(step))))));
  }

  _extractValues(val) {
    if (Array.isArray(val)) {
      // double cursor
      return {
        value: this._fixValue(val[0]),
        value2: (val.length > 1 && !isNaN(val[1]))
          ? this._fixValue(val[1])
          : this._fixValue(val[0])
      }
    } else {
      // single cursor
      return {
        value: this._fixValue(val)
      }
    }
  }

  _doSetValues(values) {
    if (this.properties.value !== values.value && !isNaN(values.value)) {
      // eslint-disable-next-line ember/no-assignment-of-untracked-properties-used-in-tracking-contexts
      this.properties.value = values.value;
    }

    if (hasOwnProperty(values, 'value2') && this.properties.value2 !== values.value2 && !isNaN(values.value2)) {
      // eslint-disable-next-line ember/no-assignment-of-untracked-properties-used-in-tracking-contexts
      this.properties.value2 = values.value2;
    }
  }

  _doSetSlidingValues(values) {
    if (this.slidingValue !== values.value && !isNaN(values.value)) {
      this.slidingValue = values.value;
    }

    if (hasOwnProperty(values, 'value2') && this.slidingValue2 !== values.value2 && !isNaN(values.value2)) {
      this.slidingValue2 = values.value2;
    }
  }

  _updateValues(values) {
    if (this.properties.isSlideValued) {
      this._doSetValues(values);
    }

    this._doSetSlidingValues(values);

    /**
     * Se produit lorsque le curseur glisse.
     *
     * @event onCursorUpdate
     * @memberof Actor.Input.Cursor#
     * @param {Event.BaseContext} context Contexte de l'évènement.
     */
    this.runEvent('onCursorUpdate');
  }

  _setValues(values) {
    this._doSetValues(values);

    /**
     * Se produit lorsque le curseur change de position.
     *
     * @event onCursorSet
     * @memberof Actor.Input.Cursor#
     * @param {Event.BaseContext} context Contexte de l'évènement.
     */
    this.runEvent('onCursorSet');
  }

  _doStyling() {
    super._doStyling();

    this.actor.style.setProperty('--barColor', this.properties.barColor);
    this.actor.style.setProperty('--barBgColor', this.properties.barBgColor);
  }

  @action
  _onUpdateValues(val) {
    throttle(this, this._updateValues, this._extractValues(val), 100, true);
  }

  @action
  _onSetValues(val) {
    const values = this._extractValues(val);
    this._setValues(values);
    this._updateValues(values);
  }

}

Cursor.registerInDefinitions({ type: 'input/cursor', componentName: 'synapps-ui/actors/input/cursor', prototypeSetup });
