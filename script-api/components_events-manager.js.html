import { get, notifyPropertyChange, set, setProperties } from '@ember/object';
import { isEmpty, isNone, isPresent, typeOf } from '@ember/utils';
import logger from 'logger';
import { ScriptError } from 'synapps-ui/errors';
import hasOwnProperty from 'synapps-ui/tools/has-own-property';

/**
 * Les évènements
 * @namespace Event
 */

export default class EventManager {

  static setupContext(event, objectUi, args) {
    if (event.isV1) {
      // logger.debug('event v1', event.ID, args);
      return {
        ...args,
        host: args.dataStores.redy.host._eventContextV1,
        session: args.dataStores.redy.host.session._eventContextV1,
        target: objectUi._eventContextV1,
        synoStage: args.stage._eventContextV1,
        synapp: args.synapp._eventContextV1,
        value: args.value,
        parentSynoStage: objectUi.stage && objectUi.stage.stage ? objectUi.stage.stage._eventContextV1 : null,
      };
    }
    /**
     * Contexte de base d'un évènement.
     * @class BaseContext
     * @memberof Event
     */
    return {
      version() { return 2; },

      /**
       * Nom de l'évènement.
       * @member Event.BaseContext#eventName
       * @readonly
       * @type {String}
       */
      eventName: event.eventName,

      /**
       * Acteur à l'origine de l'évènement.
       * @member Event.BaseContext#target
       * @readonly
       * @type {Actor.BaseActor}
       */
      target: objectUi,

      /**
       * Scène ou composite portant l'acteur émetteur de l'évènement.
       * @member Event.BaseContext#stage
       * @readonly
       * @type Stage
       */

      /**
       * Première scène parente de l'acteur émetteur de l'évènement.
       * @member Event.BaseContext#scene
       * @readonly
       * @type Scene
       */

      /**
       * Scène racine de l'acteur.
       * @member Event.BaseContext#rootScene
       * @readonly
       * @type {Scene}
       */

      /**
       * Synapp
       * @member Event.BaseContext#synapp
       * @readonly
       * @type {Synapp}
       */

      /**
       * Magasins de données.
       * @member Event.BaseContext#dataStores
       * @readonly
       * @type {Object}
       */
      ...args,


      /**
       * Obtient l'acteur dans la même scène ou le même composite suivant sa clé.
       * @function Event.BaseContext#getActor
       * @param {string} key Clé de l'acteur à chercher.
       * @returns {Actor.BaseActor} L'acteur demandé.
       */
      getActor(key) {
        return args.stage.getActor(key);
      },

      /**
       * Méthodes utilitaires pour les scripts.
       * @member Event.BaseContext#utils
       * @readonly
       * @type {Event.BaseContextUtils}
       */

      /**
       * Méthodes utilitaires pour les scripts.
       * <br> Utilisé dans {@link Event.BaseContext.utils}.
       * @class Event.BaseContextUtils
       * @hideconstructor
       */
      utils: {

        /**
         * La valeur est elle vide ?
         * @memberof Event.BaseContextUtils
         * @param {any} value la valeur à tester
         * @return {Boolean}
         */
        isEmpty(value) {
          return isEmpty(value)
        },

        /**
         * La valeur est elle null/vide ?
         * @memberof Event.BaseContextUtils
         * @param {any} value la valeur à tester
         * @return {Boolean}
         */
        isNone(value) {
          return isNone(value)
        },

        /**
         * La valeur est elle présente ?
         * @memberof Event.BaseContextUtils
         * @param {any} value la valeur à tester
         * @return {Boolean}
         */
        isPresent(value) {
          return isPresent(value)
        },

        /**
         * Obtient la valeur d'un objet suivant un chemin.
         *         *
         * @memberof Event.BaseContextUtils
         * @param {Object} obj L'objet
         * @param {Object} path Le chemin
         * @return {any}
         */
        get(obj, path) {
          return get(obj, path);
        },

        /**
         * Assigne la valeur à un objet suivant un chemin.

         * @memberof Event.BaseContextUtils

        set(obj, path, value) {
          set(obj, path, value);

  static setupEvent(objectUi, eventName, events, eventScriptLines) {
    eventName = eventName.replace(/-/g, '/');
    if (typeOf(eventScriptLines) === 'array') {
      const event = {
        ID: `http://synapps-events/${objectUi.uniqPath}/${eventName}`,
        fx: null,
        isV1: eventScriptLines[0] === '//# script/v1',
        text: eventScriptLines.join('\n'),
        eventName,
      };
      try {
        let fx = () => {};
        fx = new Function('context', `${event.text}\n//# sourceURL=${event.ID}.js`);
        event.fx = fx;
        events[eventName] = event;
      } catch (error) {
        logger.error('Script: can not create function script', event.ID, error.message);
      }

      return event;
    } else if (typeOf(eventScriptLines) === 'function') {
      const event = {
        ID: `http://synapps-events/${objectUi.uniqPath}/${eventName}`,
        fx: null,
        eventName,
      };
      try {
        let fx = () => {};
        fx = eventScriptLines;
        event.fx = fx;
        events[eventName] = event;
      } catch (error) {
        logger.error('Script: can not create function script', event.ID, error.message);
      }

      return event;
    }
  }

  static setupEvents(objectUi, eventSetups) {
    var events = {};
    Object.keys(eventSetups).forEach(key => {
      if (!eventSetups[key]) return;
      this.setupEvent(objectUi, key, events, eventSetups[key]);
    });
    return events;
  }

  static hasEvent(objectUi, eventName) {
    return objectUi.events && hasOwnProperty(objectUi.events, eventName);
  }

  static runEvent(objectUi, eventName, args) {
    if (this.hasEvent(objectUi, eventName)) {
      try {
        const event = objectUi.events[eventName];
        return event.fx.bind(objectUi)(this.setupContext(event, objectUi, args));
      } catch (e) {
        const error = new ScriptError();
        error.message = `Error during ${eventName} with ${objectUi.key}.`;
        error.details = e;
        logger.error(error.message, e);
        throw error;
      }
    }

    return true;
  }

}
