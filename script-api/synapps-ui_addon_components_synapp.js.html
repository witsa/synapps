  constructor(owner, args) {
    super(owner, args);
    assert('[Synapp][constructor] setup is required.', args.setup);

    runInDebug(() => {
      this._isInDebug = true;
    });

    this._backstageSetup = {
      key: 'backupStage',
      leadActor: JSON.stringify({
        key: 'root',
        type: 'layout/stack',
        children: args.setup.dataSourceSetups
      })
    };

    if (args.setSynapp) {
      args.setSynapp(this);
    }
  }

  @action
  _onIncludeLoad(include) {
    include.isLoaded = true;
    logger.debug('_onIncludeLoad', include.fileName);
  }

  @action
  _onIncludeLoadError(include, error) {
    include.isLoaded = false;
    include.error = error;
    logger.error('_onIncludeLoadError', include.fileName, error);
  }

  _iterateJsFileSetup(element) {
    let fileSetup = this._jsIncludesIterator.next().value;
    if (!fileSetup) {
      logger.debug('isIncludesLoaded');
      this._isIncludesLoaded = true;
      return;
    }

    if (fileSetup.isRemote) {
      const script = document.createElement('script');
      script.src = fileSetup.url;
      script.addEventListener('load', () => {
        this._onIncludeLoad(fileSetup);
        this._iterateJsFileSetup(element);
      });
      script.addEventListener('error', error => {
        this._onIncludeLoadError(fileSetup, error);
        this._iterateJsFileSetup(element);
      });
      element.append(script);
    } else {
      try {
        const ID = `http://synapps-includes/${fileSetup.fileName}`;
        const fx = new Function(`${fileSetup.content}\n//# sourceURL=${ID}`);
        fx();
        element.append(document.createComment(`includes - ${ID}`));
        this._onIncludeLoad(fileSetup);
      } catch (error) {
        this._onIncludeLoadError(fileSetup, error);
      } finally {
        this._iterateJsFileSetup(element);
      }
    }
  }

  @action
  _onDidInsertJsIncludesElement(element) {
    this._iterateJsFileSetup(element);
  }

  @action
  _onBackstageLeadActorDidRender(leadActor) {
    this._backstage = leadActor.stage;
    this._isBackstageReady = true;
  }

  @action
  _onWillRemoveComponent() {
    if (this.args.setSynapp) {
      this.args.setSynapp(null);
    }
  }

  willDestroy() {
    if (this.isDestroyed) return;
    this._backstage = null;
    super.willDestroy();
  }

  /**
   * Navigue vers une scène et change ses paramètres.
   * @todo Donner des exemples.
   * @method Synapps.Synapp#navigate
   * @param {String} sceneKey
   * @param {Object|String} sceneParams Paramètres de scène. On attend ici un objet clé/valeur ou un JSON en chaîne de cet objet. Les clés correspondent aux paramètres de la scène cible.
   */
  navigate(sceneKey, sceneParams) {
    // aucune navigation en preview.
    if (this.isInDesigner) return;
    if (isEmpty(sceneKey)) {
      this.router.transitionTo('synapps.synapp.index', this.setup.key);
      return;
    }

    if (isEmpty(sceneParams)) {
      this.router.transitionTo('synapps.synapp.scene', this.setup.key, sceneKey, {
        queryParams: {
          params: undefined,
          sceneProps: undefined,
        }
      });
      return;
    }

    if (typeOf(sceneParams) === 'object') {
      sceneParams = JSON.stringify(sceneParams);
    }

    this.router.transitionTo('synapps.synapp.scene', this.setup.key, sceneKey, {
      queryParams: {
        params: sceneParams,
        sceneProps: undefined,
      }
    });

  }

    const BaseClass = super._getEventContextV1Class();

      userAgent;

      displayScene(sceneKey) {
        this._objectUi.navigate(sceneKey);
      }
    };

    return Class;
  }
