import { action } from '@ember/object';
import { alias, readOnly } from '@ember/object/computed';
import { inject as service } from '@ember/service';
import { isEmpty, isNone } from '@ember/utils';
import { tracked } from '@glimmer/tracking';
import BaseActor from 'synapps-ui/components/actors/base-actor';
import prototypeSetup from 'synapps-ui/components/actors/display/screen-prototype-setup';
import { NotFoundError } from 'synapps-ui/errors';
import hasOwnProperty from 'synapps-ui/tools/has-own-property';

/**
 * Acteur Écran permettant d'afficher une scène de la synapp.
 *
 * **Type :** `display/screen`
 * @class Actor.Display.Screen
 * @extends Actor.BaseActor
 * @see Actor.BaseActor
 */
export default class Screen extends BaseActor {

  @service
  dataStores;

  get isDisplayed() {
    return isNone(this.args.isDisplayed) ? true : this.args.isDisplayed;
  }

  @tracked
  _displayedScene;

  @readOnly('_displayedScene')
  displayedScene;

  get setupStore() {
    return this.dataStores.setup;
  }

  @tracked
  _isLoading = false;
  get isLoading() {
    return this._isLoading;
  }

  @tracked
  _sceneSetup;
  get sceneSetup() {
    return this._sceneSetup;
  }


  @tracked
  _error = null;
  get error() {
    return this._error;
  }

  async _getSceneSetup() {
    if (isEmpty(this.properties.sceneKey)) return null;
    try {
      this._isLoading = true;
      this._error = null;
      const sceneSetup = await this.setupStore.loadScene({ synapp: this.scene.synapp.setup, sceneKey: this.properties.sceneKey });
      return sceneSetup;
    } catch (e) {
      this._isLoading = false;
      if (e instanceof NotFoundError) {
        this._error = e.message;
      } else {
        this._error = e.message || e;
      }

      throw e;
    }

  }

  _setSceneSetup(setup) {
    if (setup) {
      const sceneSetup = Object.assign({}, setup);
      sceneSetup.additionals = Object.assign({}, setup.additionals || {});
      Object.assign(sceneSetup.additionals, this.setup.additionals || {});
      this._sceneSetup = sceneSetup;
    } else {
      this._sceneSetup = null;
    }

    /**
     * Se produit lorsque l'acteur va changer de scène.
     *
     * Attention, les additionnelles de l'acteur ont déjà été transmises à la scène suivante.
     *
     * @event onWillSceneChange
     * @memberof Actor.Display.Screen#
     * @param {Event.BaseContext} context Contexte de l'évènement.
     */
    this.runEvent('onWillSceneChange');
  }

  _doPostInit() {
    super._doPostInit();
    this._onIsDisplayedChange();
  }

  _doAfterPropertyChanged({ propertyPath, name, prototyped, ...others }) {
    if (this.isDestroying || this.isDestroyed) return;

    if (propertyPath === 'properties.sceneKey' || propertyPath === 'properties.visible') {
      this._onIsDisplayedChange();
    }

    super._doAfterPropertyChanged({ propertyPath, name, prototyped, ...others });

    if (this.displayedScene && prototyped.containerName === 'additionals' && hasOwnProperty(this.displayedScene.setup.additionalDefs, name)) {
      this.displayedScene.additionals[name] = this.additionals[name];
    }
  }

  @action
  _onSceneReady(scene) {
    this._displayedScene = scene;
    scene._screen = this;

    /**
     * Se produit lorsque l'acteur a changé de scène et qu'elle est prête.
     *
     * @event onDidSceneChange
     * @memberof Actor.Display.Screen#
     * @param {Event.BaseContext} context Contexte de l'évènement.
     */
    this.runEvent('onDidSceneChange');
    this._isLoading = false;
  }

  @action
  async _onIsDisplayedChange() {
    this._sceneSetup = null;
    if (this.isDisplayed && this.properties.visible) {
      await this._displayScene();
    }
  }

  async _displayScene() {
    const setup = await this._getSceneSetup();
    this._setSceneSetup(setup);
  }

  willDestroy() {
    this._displayedScene = null;
    super.willDestroy();
  }

  @action
  _onSceneWillDestroy() {
    this._displayedScene = null;

     /**
     * Se produit lorsque l'acteur décharge la scène.
     *
     * @event onDidSceneDestroy
     * @memberof Actor.Display.Screen#
     * @param {Event.BaseContext} context Contexte de l'évènement.
     */
    this.runEvent('onDidSceneDestroy');
  }

  _doComputeStyle({ properties, isAlreadyWrite = false, style = {} }) {
    super._doComputeStyle({ properties, isAlreadyWrite, style });

    this.styling.setupStyleKey(style, "overflowX", properties, isAlreadyWrite);
    this.styling.setupStyleKey(style, "overflowY", properties, isAlreadyWrite);

    return style;
  }

  navigate(sceneKey, additionals) {

    if (this.properties.sceneKey !== sceneKey) {
      // on assure ici la transition de scène.
      this.properties.sceneKey = null;
    }

    if (additionals && this.additionals) {
      Object.keys(additionals.prototypeSetup).forEach(key => {
        // on assure ici la possibilité de commander des paramètres de la scene dans laquelle
        // on va naviguer sans avoir préparer l'acteur  écran avec des additionnelles correspondantes.
        // Il est donc inutile ici de verifier que l'écran contient les définitions.

        this.additionals[key] = additionals[key];

        // le pont avec le setup ne se fait pas automatiquement s'il n'y a pas de
        // définition. On le fait donc manuellement.
        this.setup.additionals[key] = additionals[key];

      });
    }

    this.properties.sceneKey = sceneKey;
  }

  // [MIGRATION V1]

  _getEventContextV1Class() {

    const BaseClass = super._getEventContextV1Class();

    return class extends BaseClass {

      @alias('_objectUi.properties.sceneKey')
      sceneLabel;

    };
  }

}

Screen.registerInDefinitions({ type: 'display/screen', componentName: 'synapps-ui/actors/display/screen', prototypeSetup });
