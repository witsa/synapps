/* eslint-disable ember/no-global-jquery */
/* eslint-disable ember/no-jquery */
import { computed, get } from '@ember/object';
import { later } from '@ember/runloop';
import { isNone } from '@ember/utils';
import { tracked } from '@glimmer/tracking';
import BaseActor from 'synapps-ui/components/actors/input/base-input';
import { textOverflowStyle } from 'synapps-ui/components/actors/input/common';
import prototypeSetup from 'synapps-ui/components/actors/input/switch-button-prototype-setup';

const isAuto = s => s === 'auto' || isNone(s);

/**
 * Acteur Bouton interrupteur.
 * Type : **input/switch-button**
 *
 * @class Actor.Input.SwitchButton
 * @extends Actor.Input.BaseInput
 */
export default class SwitchButton extends BaseActor {

  /**
   * Obtient les propriétés de l'acteur.
   * @member properties
   * @memberof Actor.Input.SwitchButton#
   * @type {Actor.Input.SwitchButtonProperties}
   * @readonly
   */

  @computed('properties.size')
  get _size() {
    let sizeClassName = '';
    switch (this.properties.size ? this.properties.size.toLowerCase() : 'default') {
      case 'mini':
      case 'small':
        sizeClassName = 'sm'
        break;

      case 'large':
        sizeClassName = 'lg'
        break;

      case 'default':
      default:
        break;
    }

    return sizeClassName;
  }

  @computed('properties.{modeTrue,outline}')
  get _modeTrue() {
    return this._mode('modeTrue');
  }

  @computed('properties.{modeFalse,outline}')
  get _modeFalse() {
    return this._mode('modeFalse');
  }

  _mode(propertyName) {
    let mode = get(this.properties, propertyName);
    if (!this.properties.prototypeSetup[propertyName].options.includes(mode)) {
      mode = 'default';
    }

    let modeName = 'light';
    switch (get(this.properties, propertyName)) {
      case 'default':
        break;

      default:
        modeName = mode;
        break;
    }

    return `${this.properties.outline ? 'outline-' : ''}${modeName}`;
  }

  @tracked
  _initialPropertiesValue = false;

  _init(setup) {
    super._init(setup);
    this._initialPropertiesValue = this.properties.value;
  }

  _doComputeLayout(...args) {
    if (this.isDestroyed || this.isDestroying) return;
    super._doComputeLayout(...args);
    later(this, this._setupSwitch, 300);
  }

  _onWillRender() {
    if (this.isDestroyed || this.isDestroying) return;
    super._onWillRender();
    this._setupSwitch();
  }

  _setupSwitch() {
    if (this.isDestroyed || this.isDestroying) return;
    // eslint-disable-next-line no-undef
    const $switch = $(this.actor.querySelector('input.switch-container'));
    $switch.bootstrapToggle('destroy');
    $switch.bootstrapToggle({
      on: this.properties.textTrue,
      off: this.properties.textFalse,
      size: this._size,
      onstyle: this._modeTrue,
      offstyle: this._modeFalse,
      width: isAuto(this.properties.width) ? null : this.properties.width,
      height: isAuto(this.properties.height) ? null : this.properties.height,
    });

    $switch.change(() => {
      if (!this._isInitialized) return;
      if (!!$switch.prop('checked') === !!this.properties.value) return;
      this.properties.value = !!$switch.prop('checked');

      /**
       * Se produit lorsque l'interrupteur est commuté.
       *
       * @event onSwitched
       * @memberof Actor.Input.SwitchButton#
       * @param {Event.BaseContext} context Contexte de l'évènement.
       */
      this.runEvent('onSwitched');
    });

    $switch.bootstrapToggle(this.properties.enabled ? 'enable' : 'disable');
  }

  _doComputeStyle(args) {
    if (this.isDestroying || this.isDestroyed) return;
    const { properties, isAlreadyWrite = false } = args;
    const style = super._doComputeStyle(args);

    const size = properties.size ? properties.size.toLowerCase() : null;
    if (!isAlreadyWrite && size !== 'default') {
      style.fontSize = '';
      style.lineHeight = '';
      style.padding = ''
    }

    textOverflowStyle.bind(this)({ properties, isAlreadyWrite, style });

    return style;
  }

  _doAfterPropertyChanged({ propertyPath, newValue, ...others }) {
    if (this.isDestroying || this.isDestroyed) return;
    super._doAfterPropertyChanged({ propertyPath, newValue, ...others });

    if (propertyPath === 'properties.value') {
      // eslint-disable-next-line no-undef
      const $switch = $(this.actor.querySelector('input.switch-container'));
      if (!this.properties.enabled) {
        // HACK: impossible de changer la valeur du composant s'il est désactivé.
        // on ici l'active temporairement.
        $switch.bootstrapToggle('enable');
        $switch.bootstrapToggle(newValue ? 'on' : 'off');
        $switch.bootstrapToggle('disable');
      } else {
        $switch.bootstrapToggle(newValue ? 'on' : 'off');
      }
    }

    if (propertyPath === 'properties.enabled') {
      // eslint-disable-next-line no-undef
      const $switch = $(this.actor.querySelector('input.switch-container'));
      $switch.bootstrapToggle(newValue ? 'enable' : 'disable');
    }

  }

}

SwitchButton.registerInDefinitions({ type: 'input/switch-button', componentName: 'synapps-ui/actors/input/switch-button', prototypeSetup });
