/* eslint-disable ember/no-empty-glimmer-component-classes */
import { assert } from '@ember/debug';
import { action, set } from '@ember/object';
import { readOnly } from '@ember/object/computed';
import { debounce, later } from '@ember/runloop';
import { dasherize } from '@ember/string';
import { isEmpty } from '@ember/utils';
import Component from '@glimmer/component';
import { tracked } from '@glimmer/tracking';
import logger from 'logger';
import SourceOfInternalBinding from 'synapps-ui/behaviors/source-of-internal-binding';
import V1EventScript from 'synapps-ui/behaviors/v1-event-script';
import WithAdditionals from 'synapps-ui/behaviors/with-additionals';
import InternalBinding from 'synapps-ui/bindings/internal';
import Stage from 'synapps-ui/components/behaviors/stage';
import CssClassesRegister from 'synapps-ui/registers/css-classes';
import hasOwnProperty from 'synapps-ui/tools/has-own-property';

const LEAD_ACTOR_PARSING_DELAY = 150;

@Stage
@SourceOfInternalBinding
@WithAdditionals
@V1EventScript
class SceneBase extends Component {}

/**
 * La scène.
 * @class Synapps.Scene
 * @extends Behavior.Stage
 * @extends Behavior.SourceOfInternalBinding
 * @extends Behavior.WithAdditionals
 * @see Behavior.Stage
 * @see Behavior.SourceOfInternalBinding
 * @see Behavior.WithAdditionals
 */
export default class Scene extends SceneBase {

  @tracked
  _key = null;

  _name = null;

  _cssClassesRegister = new CssClassesRegister(this);

  @tracked
  _classes = [];

  @tracked
  classesCount = 0;

  @tracked
  _instanceOfPath = '~';

  _synapp = null;

  @tracked
  _error = null;

  _screen;

  /**
   * Obtient éventuellement l'acteur écran qui l'accueille.
   * @member screen
   * @memberof Synapps.Scene#
   * @readonly
   * @type {Actor.Display.Screen|null}
   */
  get screen() {
    return this._screen;
  }

  /**
   * Obtient la scène racine.
   * @member rootScene
   * @memberof Synapps.Scene#
   * @readonly
   * @type {Synapps.Scene}
   */
  get rootScene() {
    return this.args.rootScene || this;
  }

  constructor(owner, args) {
    super(owner, args);

    assert('[synapps-ui][scene] setup is required.', args.setup);
    assert('[synapps-ui][scene] no key in scene definition.', args.setup.key);

    this._key = args.setup.key;

    set(this, '_name', args.setup.name);

    if (args.synapp) {
      this._synapp = args.synapp;
    }

    if (args.instanceOfPath) {
      this._instanceOfPath = args.instanceOfPath;
    }

    if (args.isBackstage) {
      return;
    }

    this._extractAdditionalsFromSetup(args.setup);

    this._doUpdateAdditionalsFromParams(args.sceneParams);

    this._onDidAdditionalsChange();

  }

  /**
   * Obtient la Synapp.
   * @member Synapps.Scene#synapp
   * @readonly
   * @type {Synapps.Synapp}
   */
  get synapp() {
    return this._synapp;
  }

  get classes() {
    return this._classes;
  }

  get cssClassesRegister() {
    return this._cssClassesRegister;
  }

  /**
   * Obtient la clé unique de la scène.
   * @member Synapps.Scene#key
   * @readonly
   * @type {String}
   */
  @readOnly('_key')
  key;

  /**
   * Obtient le nom de la scène.
   * @member Synapps.Scene#name
   * @readonly
   * @type {String}
   */
  @readOnly('_name')
  name;

  get instanceOfPath() {
    return this._instanceOfPath;
  }


  /**
   * Obtient le chemin unique de la scène.
   * @member Synapps.Scene#uniqPath
   * @readonly
   * @type {String}
   */
  get uniqPath() {
    return `${this.instanceOfPath}/${this.key}`;
  }

  /**
   * Obtient la classe CSS unique de la scène.
   * @member Synapps.Scene#uniqClass
   * @readonly
   * @type {String}
   */
  get uniqClass() {
    return this.uniqPath.replace(/~/, '').replace(/\//g, '_');
  }

  setupClass(style) {

    var str = '';
    for (var prop in style) {
      str += `\n  ${dasherize(prop)}: ${style[prop]};`;
    }

    return str;
  }

  recordClass(properties, objectUi, styleName) {
    if (this.args.isBackstage) return;
    const classSelector = this.cssClassesRegister.buildKey({ actorClass: objectUi.constructor, styleName });
    if (this.cssClassesRegister.hasKey(classSelector)) return;
    logger.debug('recordClass', classSelector);
    const css = this.setupClass(objectUi._doComputeStyle({ properties, isAlreadyWrite: true }));
    this.cssClassesRegister.record({ record: css, actorClass: objectUi.constructor, styleName });

    later(() => {
      this._classes = [...this.cssClassesRegister.register].map(([key, value]) => {
        return {
          classSelector: key,
          css: value
        };
      });
      this.styleDom.innerHTML = this.classes.map(cl => `
  ${cl.classSelector} {
    ${cl.css}
  }
`).join('');
    });

    this.classesCount = this.cssClassesRegister.register.size;
  }

  unrecordClass(actorClass, styleName) {
    if (this.args.isBackstage) return;
    const classSelector = this.cssClassesRegister.buildKey({ actorClass, styleName });
    logger.debug('unrecordClass', classSelector);
    this.cssClassesRegister.unrecord({ actorClass, styleName });

    later(() => {
      this._classes = [...this.cssClassesRegister.register].map(([key, value]) => {
        return {
          classSelector: key,
          css: value
        };
      });
    });

    this.classesCount = this.cssClassesRegister.register.size;
  }

  @action
  didInsert(element) {
    this.styleDom = element;
    if (this.args.onReady) {
      // TODO un event onReady
      this.args.onReady(this);
    }

    later(() => {
      if (this.isDestroying || this.isDestroyed) return;
      try {
        const leadActorSetup = typeof this.args.setup.leadActor === 'string' ? JSON.parse(this.args.setup.leadActor) : this.args.setup.leadActor;
        this.setLeadActorSetup(leadActorSetup);
      } catch (error) {
        logger.error('[Scene][didInsert] error during the parse of leadActorSetup.', error.message || error);
        this._error = error.message || error;
      }
    }, LEAD_ACTOR_PARSING_DELAY);
  }

  @action
  _sceneWillDestroy() {
    if (this.args.onWillDestroy) {
      this.args.onWillDestroy();
    }
  }

  willDestroy() {
    if (this.isDestroyed) return;

    this._screen = null;

    this.cssClassesRegister.destroy();
    this._cssClassesRegister = null;

    super.willDestroy();
  }

  _doAfterAdditionalChanged({ name, ...others }) {
    if (this.isDestroying || this.isDestroyed) return;
    super._doAfterAdditionalChanged({ name, ...others });

    if (this.screen && hasOwnProperty(this.screen.setup.additionalDefs, name)) {
      this.screen.additionals[name] = this.additionals[name];
      return;
    }

    debounce(this, this._onDidAdditionalsChange, 250);
  }

  _onDidAdditionalsChange() {
    if (this.isDestroying || this.isDestroyed) return;
    if (this.args.onDidAdditionalsChange) {
      this.args.onDidAdditionalsChange(this._additionalsToSceneParams(this.additionals));
    }
  }

  _additionalsToSceneParams(additionals) {
    const sceneParams = {};
    if (!additionals) return;
    const additionalsSetupEntries = Object.entries(additionals.prototypeSetup);
    for (const [key, setup] of additionalsSetupEntries) {
      if (
        (
          // ça vient d'un nav-button ou d'un screen
          additionals.objectUi !== this

          // [MIGRATION V1]
          // inhibition des additionals de migration dans le cas d'une navigation depuis un nav-button
          && (
            this.synapp.setup.setupVersion > 1
            || additionals.objectUi.type !== 'input/nav-button'
            || ![
              'relativePath',
              'datasourceVariable',
              'bootstrapColor'
            ].includes(key))
        )

        // ça vient d'un changement de paramètre de la scène et l'apparition dans l'URL est autorisée
        || setup.allowUrlParams
      ) {
        sceneParams[key] = additionals[key];
      }
    }

    const params = JSON.stringify(sceneParams);
    return params === '{}' ? '' : params;
  }

  _doUpdateAdditionalsFromParams(sceneParamsJSON) {
    sceneParamsJSON = sceneParamsJSON || this.args.sceneParams;
    if (isEmpty(sceneParamsJSON)) return;
    let sceneParams = {};
    try {
      sceneParams = JSON.parse(sceneParamsJSON);
    } catch (e) {
      logger.warn('[Scene][_doUpdateAdditionalsFromParams] impossible to parse sceneParams.', sceneParamsJSON);
    }

    Object.keys(sceneParams).forEach(key => {
      if (hasOwnProperty(this.additionals.prototypeSetup, key)) {
        this.additionals[key] = sceneParams[key];
      }
    });
  }

  @action
  _didUpdateSceneParams() {
    this._doUpdateAdditionalsFromParams();
  }

  /**
   * Navigue vers une scène et change ses paramètres.
   * @method Synapps.Scene#navigate
   * @param {String} sceneKey
   * @param {Synapps.PrototypedBase} additionals
   */
  navigate(sceneKey, additionals) {
    if (sceneKey && this.key !== sceneKey) {
      // on navigue vraiment sur une autre scene
      if (this.screen) {
        this.screen.navigate(sceneKey, additionals);
        return;
      }

      this.synapp.navigate(sceneKey, this._additionalsToSceneParams(additionals));
      return;
    }

    if (additionals && this.additionals) {
      Object.keys(additionals.prototypeSetup).forEach(key => {
        if (hasOwnProperty(this.additionals.prototypeSetup, key)) {
          this.additionals[key] = additionals[key];
        }
      });
    }
  }

  // [MIGRATION V1]

  _getEventContextV1Class() {

    const BaseClass = super._getEventContextV1Class();

    return class extends BaseClass {

      @readOnly('_objectUi.key')
      label;
    };

  }

}

InternalBinding.registerInInternalSourceDefs({
  sourceType: 'stage',
  resolveSource: function resolveSource({ target }) {
    return target.stage;
  }
});
